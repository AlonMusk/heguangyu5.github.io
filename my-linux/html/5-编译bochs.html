编译bochs


sudo apt-get install libsdl1.2-dev libgtk2.0-dev
./configure --enable-debugger --enable-debugger-gui --enable-disasm --with-sdl --enable-x86-64 --enable-smp
make
sudo make install

.bochsrc

megs: 128
display_library: sdl, options="gui_debug"
romimage: file=$BXSHARE/BIOS-bochs-latest
vgaromimage: file=$BXSHARE/VGABIOS-lgpl-latest
ata0-master: type=disk, path="my-linux.img", mode=flat, cylinders=1015, heads=16, spt=63
boot: disk

try it

bochs -q
